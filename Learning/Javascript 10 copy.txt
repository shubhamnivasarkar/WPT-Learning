            function squaresum(n){
                sum = 0
                for(i=1;i<=n;i++)
                sum+=(2 * 1) * (2 * 1);
                return sum;
            }
            
        </script>

            var t = parseInt(prompt("Enter the number :"));
            document.write(squaresum(t))
